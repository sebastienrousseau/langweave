1//! # Performance Optimizations Module
2//!
3//! This module provides optimized alternatives to core LangWeave functions:
4//! - Const static arrays instead of heap-allocated vectors
5//! - O(1) membership testing via compile-time pattern matching
6//! - Borrowed strings to eliminate clones
7//! - True zero-allocation variants for hot paths
8//!
9//! **Note:** This module is experimental. API may change in future versions.
10
11use crate::I18nError;
12
13/// Compile-time constant array of supported language codes.
14///
15/// This replaces the heap-allocated `Vec<String>` from `supported_languages()`.
16/// With only 15 languages, linear search with `eq_ignore_ascii_case` is faster
17/// than HashSet lookup due to cache locality and avoiding hash computation.
18pub const SUPPORTED_LANGUAGE_CODES: &[&str] = &[
19    "en", "fr", "de", "es", "pt", "it", "nl", "ru", "ar", "he", "hi",
20    "ja", "ko", "zh", "id",
21];
22
23/// Optimized version of supported_languages() that returns borrowed string slices
24/// instead of heap-allocated Strings.
25///
26/// Performance improvement: Eliminates 15 heap allocations per call
27///
28/// # Returns
29///
30/// A static slice of string slices containing the supported language codes.
31///
32/// # Examples
33///
34/// ```
35/// use langweave::optimized::supported_languages_optimized;
36///
37/// let languages = supported_languages_optimized();
38/// assert_eq!(languages.len(), 15);
39/// assert!(languages.contains(&"en"));
40/// assert!(languages.contains(&"fr"));
41/// ```
42pub fn supported_languages_optimized() -> &'static [&'static str] {
43    SUPPORTED_LANGUAGE_CODES
44}
45
46/// Optimized version of is_language_supported() with zero heap allocations.
47///
48/// Uses compile-time pattern matching for O(1) lookup without any allocations.
49/// Supports case-insensitive matching via ASCII case folding (no heap allocation).
50///
51/// # Arguments
52///
53/// * `lang` - A string slice that holds the language code to validate.
54///
55/// # Returns
56///
57/// `true` if the language is supported, `false` otherwise.
58///
59/// # Examples
60///
61/// ```
62/// use langweave::optimized::is_language_supported_optimized;
63///
64/// assert!(is_language_supported_optimized("en"));
65/// assert!(is_language_supported_optimized("FR")); // Case insensitive
66/// assert!(!is_language_supported_optimized("zz"));
67/// ```
68pub fn is_language_supported_optimized(lang: &str) -> bool {
69    // Zero-allocation case-insensitive check using eq_ignore_ascii_case
70    SUPPORTED_LANGUAGE_CODES
71        .iter()
72        .any(|&code| code.eq_ignore_ascii_case(lang))
73}
74
75/// True zero-allocation language support check.
76///
77/// This function uses compile-time pattern matching for lowercase codes (fast path),
78/// and falls back to ASCII case-insensitive comparison without any heap allocation.
79///
80/// # Arguments
81///
82/// * `lang` - A string slice that holds the language code to validate.
83///
84/// # Returns
85///
86/// `true` if the language is supported, `false` otherwise.
87///
88/// # Examples
89///
90/// ```
91/// use langweave::optimized::is_language_supported_zero_alloc;
92///
93/// assert!(is_language_supported_zero_alloc("en"));
94/// assert!(is_language_supported_zero_alloc("FR")); // Case insensitive, zero alloc
95/// assert!(!is_language_supported_zero_alloc("zz"));
96/// ```
97#[inline]
98pub fn is_language_supported_zero_alloc(lang: &str) -> bool {
99    // Fast path: exact lowercase match (most common case)
100    match lang {
101        "en" | "fr" | "de" | "es" | "pt" | "it" | "nl" | "ru"
102        | "ar" | "he" | "hi" | "ja" | "ko" | "zh" | "id" => true,
103        _ => {
104            // Slow path: case-insensitive check without allocation
105            // Uses eq_ignore_ascii_case which operates in-place
106            SUPPORTED_LANGUAGE_CODES
107                .iter()
108                .any(|&code| code.eq_ignore_ascii_case(lang))
109        }
110    }
111}

113/// Translation function with optimized language validation.
114///
115/// Uses the original translate function with pre-validated language support.
116/// Note: Translation itself requires string allocation for the result.
117///
118/// # Arguments
119///
120/// * `lang` - A string slice that holds the target language code (e.g., "en", "fr").
121/// * `key` - A string slice that holds the key to be translated.
122///
123/// # Returns
124///
125/// * `Ok(String)` - The translated text.
126/// * `Err(I18nError)` - An error if the translation fails.
127///
128/// # Examples
129///
130/// ```
131/// use langweave::optimized::translate_optimized;
132///
133/// let result = translate_optimized("fr", "Hello");
134/// assert!(result.is_ok());
135/// ```
136///
137/// # Errors
138///
139/// This function will return an error if:
140/// * The specified language is not supported.
141/// * The translation key is not found in the language's translation dictionary.
142pub fn translate_optimized(
143    lang: &str,
144    key: &str,
145) -> Result<String, I18nError> {
146    // Direct call to the underlying translation system
147    crate::translations::translate(lang, key)
148}
